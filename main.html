Endpoints:
POST | GET /api/v1
Takes a screenshot of a website and returns an Imgur URL.
Headers:
{"website": "URL"}
Returns:
{"snapshot": "URL", "website": "URL", "status": 200}
This is an
example
for the discord.py@rewrite library 

from discord.ext import commands
import aiohttp

@bot.command()
@commands.cooldown(1, 5, commands.BucketType.guild)
async def ss(self, ctx, *, url: str):
    async with ctx.typing(), aiohttp.ClientSession() as session:
        screener = "http://webscreener.herokuapp.com/api/v1"
        async with session.post(screener, headers=dict(website=url)) as r:
            website = (await r.json())["snapshot"]
            await ctx.send(embed=discord.Embed(color=discord.Color.blurple()).set_image(url=website))
